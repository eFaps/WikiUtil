\documentclass[final,a4paper,10pt,oneside]{book}
\usepackage{xcolor,graphics,a4,lastpage,setspace}

% to get correct official euro symbol
\usepackage{eurosym}

% tables on multiple sides with automatic width columns
\usepackage{tabularx}
\usepackage{longtable}
\usepackage{ltxtable}

% To embed file content directly in a tex file
\usepackage{filecontents}

% PDF Support
\usepackage[pdftex]{graphicx}

% better Font for PDF
\usepackage{times}

% create PDF Thumbnails
\usepackage{thumbpdf}

% that multi page PDF file could be included
\usepackage{pdfpages}

% Hyperlinks, with internal links in black color (instead of red box)
\usepackage[pdftex,colorlinks=true,linkcolor=black]{hyperref}

% Listings
\usepackage{listings}
\lstset{basicstyle=\ttfamily\fontsize{9}{12}\selectfont,xleftmargin=5mm,breaklines=true}

% paragraph indention set to 0
\setlength{\parindent}{0cm}

% change width of the text for better source code
\setlength{\oddsidemargin}{.5cm}
\setlength{\textwidth}{16cm}

% read all variables
\input{variables}

% to include preformatted text (like the license file)
\usepackage{fancyvrb}

% to define appendix of the book
\usepackage{appendix}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{document}

    % define the PDF properties
    \ifdefined\WikiAuthor
        \hypersetup{pdfauthor=\WikiAuthor}
    \fi
    \ifdefined\WikiTitle
        \hypersetup{pdftitle=\WikiTitle}
    \fi
    \ifdefined\WikiSubTitle
        \hypersetup{pdfsubject=\WikiSubTitle}
    \fi
    \ifdefined\WikiPDFCreator
        \hypersetup{pdfcreator=\WikiPDFCreator}
    \else
        \hypersetup{pdfcreator=eFaps Wiki Util}
    \fi
    \ifdefined\WikiPDFKeywords
        \hypersetup{pdfkeywords=\WikiPDFKeywords}
    \fi

    % import front page (title page)
    \input{frontpage}

    \tableofcontents

    \newpage

    \input{content}

    \ifdefined\WikiLicenseFile
        \appendixpage
        \addcontentsline{toc}{part}{Appendices}
        \appendix
        \chapter{License}
        \VerbatimInput[baselinestretch=1,fontsize=\footnotesize,numbers=none]{\WikiLicenseFile}
    \fi

\end{document}
